       async function createLayer(key){
    try {
        const url = FILES[key];
        const buf = await fetchDecode(url);

        const src = geCtx.createBufferSource();
        src.buffer = buf;
        src.loop = true;
        src._GE90_TAG = true;

        g._GE90_TAG = true;

        lp.type = 'lowpass';
        lp.frequency.value =
            key === 'idle' ? 1200 :
            key === 'n1'   ? 2000 :
                             900;

            togaHF.frequency.value = 20000;
            togaHF.Q.value = 0.7;

            // Correct chain: src → g → togaHF → lp → master
            src.connect(g);
            g.connect(togaHF);
            togaHF.connect(lp);
        } else {
            // Normal chain for idle and n1
            src.connect(g);
            g.connect(lp);
        }

        lp.connect(master);
        src.start(0);

        window.GE90_layers = window.GE90_layers || {};
        window.GE90_layers[key] = { src, gainNode: g, filter: lp, togaHF };

          // -------------------------
          // Throttle curve
          // -------------------------
          function computeTargets(t){
            const clamp = v => Math.max(0, Math.min(1, v));

            const idleCut = 0.45;
            const n1Start = 0.12, n1Full = 0.52;
            const togaStart = 0.42;
            const MID_CENTER = 0.33, MID_WIDTH = 0.32, MID_GAIN = 2.0;
            const MIN_LAYER_FLOOR = 0.1;

            const rawIdleLinear = clamp(1 - (t / idleCut));
            const idleRaw = Math.pow(rawIdleLinear, IDLE_POWER);

            let n1Raw = 0;
            if (t > n1Start)
              n1Raw = t <= n1Full ? (t - n1Start) / (n1Full - n1Start) : 1;

            const togaRaw = t <= togaStart ? 0 : Math.pow((t - togaStart) / (1 - togaStart), 1.15);


            const dx = Math.max(0, 1 - Math.abs((t - MID_CENTER) / (MID_WIDTH/2)));
            const bell = 1 + (MID_GAIN - 1) * Math.pow(dx, 1.6);

            const idleBoosted = clamp(idleRaw * (1 + (bell - 1) * 0.65));
            const n1Boosted   = clamp(n1Raw * bell);
            const togaBoosted = Math.pow(togaRaw, 1.05);



            const idleFinalRaw = Math.max(idleBoosted, MIN_LAYER_FLOOR * 1.25) * IDLE_VOLUME_BOOST;
            const idleFinal    = Math.min(MAX_LAYER_GAIN, idleFinalRaw);

            const n1Final   = Math.max(n1Boosted * N1_SCALE, MIN_LAYER_FLOOR * 0.95);
            const togaFinal = togaBoosted;
           const idleRate = 1 + (IDLE_PITCH_INTENSITY * Math.sqrt(Math.max(0, t * 9))) * 0.9;
            const n1Rate   = 1 + 0.22 * t;
            const togaRate = 1 + 0.50 * Math.pow(t, 1.05);



            return {
              gains: { idle: idleFinal, n1: n1Final, toga: Math.min(1, togaFinal) },
              rates: { idle: idleRate, n1: n1Rate, toga: togaRate }
            };
          }

          function setGainSmooth(param, target){
            const now = geCtx.currentTime;
            param.cancelScheduledValues(now);
            param.setTargetAtTime(target, now, 0.04);
          }

// Altitude‑based TOGA high‑frequency attenuation

window.applyTogaAltitudeHF = function (layer, geCtx) {
    const alt = (window.geofs?.animation?.values?.altitude) || 0;

    const maxAlt = 8000;
    const f = 1 - (alt / maxAlt);
    const altFactor = Math.max(0, Math.min(1, f));

    const minHz = 250;
    const maxHz = 20000;
    const hz = minHz + (maxHz - minHz) * Math.pow(altFactor, 1.4);

    if (layer.togaHF) {
        layer.togaHF.frequency.setTargetAtTime(hz, geCtx.currentTime, 0.15);
    }
};

// GE90_applyRawThrottle — includes wing2 attenuation + TOGA HF altitude shaping
// -------------------------------------------------------
window.GE90_applyRawThrottle = function(raw){

    const t = Math.max(0, Math.min(1, Number(raw) || 0));
    const targets = computeTargets(t);

    const view = window._GE90_viewType || "exterior";
    const wing2Factor = (view === "wing2") ? WING2_ENGINE_ATTEN : 1.0;

    ['idle','n1','toga','buzzsaw'].forEach(k=>{
        const layer = window.GE90_layers?.[k];
        if (!layer) return;

        // -----------------------------
        // BUZZSAW LOGIC (A320/A350 only)
        // -----------------------------
        if (k === "buzzsaw") {
            const buzz = window.GE90_layers?.buzzsaw;
            if (!buzz) return;

            const start = 0.70;
            const end   = 1.00;

            let amt = 0;
            if (t > start) {
                amt = (t - start) / (end - start);
                amt = Math.min(1, Math.max(0, amt));
                amt = Math.pow(amt, 1.4);
            }

            const buzzGain = amt * 0.00; // muted for B777
            buzz.gainNode.gain.setTargetAtTime(buzzGain, geCtx.currentTime, 0.08);

            if (buzz.src?.playbackRate) {
                const rate = 0.70 + amt * 0.17;
                buzz.src.playbackRate.setTargetAtTime(rate, geCtx.currentTime, 0.05);
            }

            return;
        }

        // -----------------------------
        // NORMAL ENGINE LAYERS (idle, n1, toga)
        // -----------------------------
        let g = targets.gains[k] * (window._GE90_master_boost || DEFAULT_MASTER_BOOST);

        if (k === "toga") {
            g *= 0.40;

            // ---------------------------------------------------------
            // NEW: Altitude‑based TOGA high‑frequency attenuation
            // ---------------------------------------------------------
            if (typeof window.applyTogaAltitudeHF === "function") {
                window.applyTogaAltitudeHF(layer, geCtx);
            }
        }

        g *= wing2Factor;
        g = Math.max(0, Math.min(1.3, g));

        setGainSmooth(layer.gainNode.gain, g);

   if (layer.src?.playbackRate) {
    const proximityPitch = window._GE90_proximityPitchMult || 1.0;
    layer.src.playbackRate.setTargetAtTime(targets.rates[k] * proximityPitch, geCtx.currentTime, 0.04);
}
    });
};


          // -------------------------
          // RPM bind — hysteresis
          // -------------------------
          window.GE90_enableRpmBind = function(){
            try {
              const engine = window.geofs?.aircraft?.instance?.engine;
              if (!engine) return console.log('[GE90] engine not ready');

              if (window._ge90_hardwire_poll)
                clearInterval(window._ge90_hardwire_poll);

              const initialRpm = Number(engine.rpm) || 0;
              window._GE90_lastRpm = initialRpm;
              window._GE90_engineOff = initialRpm <= RPM_SHUTDOWN_THRESHOLD;
              window._GE90_engineOffPlayed = window._GE90_engineOff;
              window._GE90_spawnTime = performance.now(); // Track spawn time

              window._ge90_hardwire_poll = setInterval(()=>{
                try {
                  const raw = Number(engine.rpm);
                  if (!isFinite(raw)) return;

                  const rpm = Math.max(0, raw);
                  const wasOff = window._GE90_engineOff;
                  const timeSinceSpawn = performance.now() - window._GE90_spawnTime;
                  const spawnProtectionActive = timeSinceSpawn < SPAWN_PROTECTION_MS;

                  // Check altitude to prevent shutdown sound when spawning in air
                  let altitudeProtectionActive = false;
                  try {
                    const altitude = window.geofs?.aircraft?.instance?.altitude || 0;
                    altitudeProtectionActive = altitude > ALTITUDE_PROTECTION_FT;
                  } catch(e) {
                    // If altitude check fails, assume no protection
                  }

                  const nowOff = rpm <= RPM_SHUTDOWN_THRESHOLD;
                  const nowOn  = rpm >= RPM_START_THRESHOLD;

                  if (window._GE90_startupSrc && rpm >= RPM_MIN) {
                    try { window._GE90_startupSrc.stop(); } catch(e){}
                    window._GE90_startupSrc = null;
                  }

                  if (wasOff && nowOn && window._GE90_lastRpm < RPM_START_THRESHOLD) {
                    // Quieter startup + fade-in of engine layers
                    playStartupDual();

                    window._GE90_engineOffPlayed = false;
                    window._GE90_engineOff = false;
                    window._GE90_lastRpm = rpm;

                    ['idle','n1','toga','buzzsaw'].forEach(k=>{

                      const layer = window.GE90_layers?.[k];
                      if (!layer) return;

                      try {
                        layer.gainNode.gain.cancelScheduledValues(geCtx.currentTime);
                        layer.gainNode.gain.setValueAtTime(0, geCtx.currentTime);
                        layer.gainNode.gain.setTargetAtTime(0.001, geCtx.currentTime, 0.25);
                      } catch(e){}
                    });
                  }

                  else if (!wasOff &&

        try { src.disconnect(); } catch(e){}

    _rLoop.mid.gain.gain.cancelScheduledValues(now);

  }

  // -------------------------
  // Accel computation (hoisted)
  // -------------------------
  function _computeAccel(){
    const now = performance.now();
    const speedKts = _readGroundspeedKts();
    if (speedKts === null) return null;
    if (_lastSpeedKts === null) {
      _lastSpeedKts = speedKts;
      _lastSpeedTs = now;
      return { speedKts, accel: 0 };
    }
    const dt = Math.max(1, now - _lastSpeedTs) / 1000;
    const ds_kts = speedKts - _lastSpeedKts;
    const ds_mps = ds_kts * 0.514444;
    const accel = ds_mps / dt;
    _lastSpeedKts = speedKts;
    _lastSpeedTs = now;
    return { speedKts, accel };
  }

          isGrounded = groundContactRaw;
        } else {
          // fallback conservative test
          isGrounded = _isOnGroundFallback();
        }

        // If not grounded, stop immediately
        if (!isGrounded) {
          _stopLoop();
          return;
        }

        // If grounded, continue with spawn protection and intensity logic
        const now = performance.now();
        const sinceSpawn = (window._GE90_spawnTime) ? (now - window._GE90_spawnTime) : 999999;
        if (sinceSpawn < SPAWN_PROTECT_MS) return;

        const s = _computeAccel();
        if (!s) return;
        const speed = s.speedKts;
        const accel = s.accel;

        // Start/Update logic with gradual introduction
        if (speed >= SPEED_START_KTS && (accel >= ACCEL_START || speed >= SPEED_MIN_KTS)) {
          if (_accelStartTime === 0) _accelStartTime = now;
          if (now - _accelStartTime >= ACCEL_HOLD_MS) {
            const intensity = _computeIntensity(speed, accel);
            if (!_rLoop) {
              await _loadAll();
              _startLoop(intensity * 0.45 + 0.01);
            } else {
              _updateLoopIntensity(intensity, LOOP_FADE);
            }
            return;
          }
        } else {
          _accelStartTime = 0;
        }

        // Stop condition: require both speed and accel low to stop
        if (_rLoop && (speed <= SPEED_STOP_KTS && accel <= ACCEL_STOP)) {
          _stopLoop();
        }
      } catch(e){
        console.warn('[GE90 Rattle] poll error', e);
      }
    }, POLL_MS);
    console.log('[GE90 Rattle] polling started (groundContact-driven)');
  }

  function _computeIntensity(speedKts, accel){
    const SPEED_FULL = 100;
    const speedBaseline = Math.max(0, Math.min(1, (speedKts - SPEED_START_KTS) / (SPEED_FULL - SPEED_START_KTS)));
    const accelIntensity = Math.max(0, Math.min(1, (accel - ACCEL_START) / (2.0 - ACCEL_START)));
    return Math.max(speedBaseline * 0.9, accelIntensity * 1.0);
  }

      const inst = window.geofs?.aircraft?.instance;
      if (!inst) return null;
      const engine = inst.engine || inst.engines?.[0] || null;
      if (!engine) return null;

      // Candidate boolean/numeric properties

        engine.on,
        engine.running,
        engine.isRunning,
        engine.master,
        engine.starter,
        engine.starterOn,
        engine.runningState,
        engine.state
      ];

      for (const c of candidates) {
        if (typeof c === 'boolean') return c;
        if (typeof c === 'number') return c > 0;
        if (typeof c === 'string' && (c === 'on' || c === 'running' || c === '1')) return true;
      }

      // Fallback: if rpm exists, treat rpm > small threshold as running
      if (typeof engine.rpm === 'number' && !isNaN(engine.rpm)) {
        return engine.rpm > 50;
      }

      return null;

// CAMERA PROXIMITY EFFECT (volume + tone + pitch shift based on camera-to-aircraft distance)
// Unified ECEF distance calculation - only active in chase/free cam
// Independent gain + filter stage per layer - bypasses throttle clamps entirely
// -------------------------------------------------------------
(function setupProximityEffect(){
    const ctx = window._GE90_audioCtx;
    if (!ctx) return;

    // Tunables
    const POLL_MS           = 100;
    const SMOOTH_ALPHA      = 0.15;

    // Default/starting distance (follow cam default) - prevents quiet startup
    const DEFAULT_DIST      = 64;

    // Distance range to map (meters) - DIST_NEAR set so default (~64m) sits at the
    // "normal" plateau (t=0), avoiding the "need to zoom in for normal volume" issue
    const DIST_NEAR         = 70;
    const DIST_FAR          = 800;

    // Volume multiplier range
    const VOL_NEAR          = 1.40;
    const VOL_FAR           = 0.20;

    // Low-pass filter frequency range
    const LPF_NEAR_HZ       = 20000;
    const LPF_FAR_HZ        = 500;

    // Pitch multiplier range (strong distance-based pitch shift)
    const PITCH_NEAR        = 1.00;
    const PITCH_FAR         = 0.82;

    // Use logarithmic distance mapping (set true if zoom feels exponential)
    const USE_LOG_MAPPING   = true;

    let smoothedDist = DEFAULT_DIST;
    let lastMode = null;

    window._GE90_proximityVolMult = 1.0;
    window._GE90_proximityLpfHz = LPF_NEAR_HZ;
    window._GE90_proximityPitchMult = 1.0;

    function clamp01(v){ return Math.max(0, Math.min(1, v)); }

    // -----------------------------------------------------------
    // WGS84 geodetic -> ECEF conversion
    // -----------------------------------------------------------
    function llaToEcef(lat, lon, alt){
        const a = 6378137;
        const e2 = 0.00669437999014;
        const latR = lat * Math.PI/180;
        const lonR = lon * Math.PI/180;
        const N = a / Math.sqrt(1 - e2 * Math.sin(latR)**2);
        const x = (N + alt) * Math.cos(latR) * Math.cos(lonR);
        const y = (N + alt) * Math.cos(latR) * Math.sin(lonR);
        const z = (N * (1-e2) + alt) * Math.sin(latR);
        return { x, y, z };
    }

    // -----------------------------------------------------------
    // Unified distance: camera (Cesium ECEF) to aircraft (LLA -> ECEF)
    // -----------------------------------------------------------
    function readCameraDistance(){
        try {
            const camPos = window.geofs?.camera?.cam?.position;
            const lla = window.geofs?.aircraft?.instance?.llaLocation;
            if (!camPos || !lla) return null;

            const acEcef = llaToEcef(lla[0], lla[1], lla[2]);
            const dx = camPos.x - acEcef.x;
            const dy = camPos.y - acEcef.y;
            const dz = camPos.z - acEcef.z;
            const dist = Math.sqrt(dx*dx + dy*dy + dz*dz);

            return isFinite(dist) ? dist : null;
        } catch(e){
            return null;
        }
    }

    // -----------------------------------------------------------
    // Wire proximity gain + LPF per layer. Retry-safe for async load.
    // Chain becomes: layer.filter -> pGain -> pLpf -> master
    // -----------------------------------------------------------
    function ensureProximityNodes(){
        if (!window._GE90_proximityNodes) window._GE90_proximityNodes = {};
        const nodes = window._GE90_proximityNodes;

        ['idle','n1','toga'].forEach(k => {
            if (nodes[k]) return;

            const layer = window.GE90_layers?.[k];
            if (!layer || !layer.gainNode || !layer.filter) return;

            try {
                const pGain = ctx.createGain();
                pGain.gain.value = 1.0;
                pGain._GE90_TAG = true;

                const pLpf = ctx.createBiquadFilter();
                pLpf.type = 'lowpass';
                pLpf.frequency.value = LPF_NEAR_HZ;
                pLpf.Q.value = 0.7;
                pLpf._GE90_TAG = true;

                try { layer.filter.disconnect(); } catch(e){}
                layer.filter.connect(pGain);
                pGain.connect(pLpf);
                pLpf.connect(window._GE90_master);

                nodes[k] = { gain: pGain, lpf: pLpf };
                console.log("[GE90 Proximity] wired gain+filter for", k);
            } catch(e){
                console.warn("[GE90 Proximity] wiring failed for", k, e);
            }
        });

        return nodes;
    }

    function applyProximity(){
        try {
            const nodes = ensureProximityNodes();
            const targetHz = window._GE90_proximityLpfHz || LPF_NEAR_HZ;
            const targetVol = window._GE90_proximityVolMult || 1.0;

            Object.values(nodes).forEach(n => {
                if (n.lpf?.frequency) n.lpf.frequency.setTargetAtTime(targetHz, ctx.currentTime, 0.15);
                if (n.gain?.gain) n.gain.gain.setTargetAtTime(targetVol, ctx.currentTime, 0.15);
            });
        } catch(e){}
    }

    setInterval(() => {
        try {
            const mode = window.geofs?.camera?.currentModeName;

            if (mode !== lastMode) {
                lastMode = mode;
                smoothedDist = DEFAULT_DIST;
            }

            // Only apply proximity effect in chase/free - follow stays neutral
            if (mode !== 'chase' && mode !== 'free') {
                window._GE90_proximityVolMult = 1.0;
                window._GE90_proximityLpfHz = LPF_NEAR_HZ;
                window._GE90_proximityPitchMult = 1.0;
                applyProximity();
                return;
            }

            const rawDist = readCameraDistance();
            if (typeof rawDist !== 'number' || !isFinite(rawDist) || rawDist <= 0) {
                // Hold last known values instead of snapping to neutral
                applyProximity();
                return;
            }

            smoothedDist += (rawDist - smoothedDist) * SMOOTH_ALPHA;

            let t;
            if (USE_LOG_MAPPING) {
                const logNear = Math.log(Math.max(1, DIST_NEAR));
                const logFar = Math.log(Math.max(2, DIST_FAR));
                const logDist = Math.log(Math.max(1, smoothedDist));
                t = clamp01((logDist - logNear) / (logFar - logNear));
            } else {
                t = clamp01((smoothedDist - DIST_NEAR) / (DIST_FAR - DIST_NEAR));
            }

            const volMult = VOL_NEAR + (VOL_FAR - VOL_NEAR) * t;
            const lpfHz = LPF_NEAR_HZ + (LPF_FAR_HZ - LPF_NEAR_HZ) * t;
            const pitchMult = PITCH_NEAR + (PITCH_FAR - PITCH_NEAR) * t;

            window._GE90_proximityVolMult = volMult;
            window._GE90_proximityLpfHz = lpfHz;
            window._GE90_proximityPitchMult = pitchMult;

            applyProximity();

        } catch(e){
            // On error, hold last values rather than reset
            applyProximity();
        }
    }, POLL_MS);

    console.log("[GE90] Camera proximity effect initialized (chase/free only, follow stays neutral)");
})();

// Smooth final master gain
let _GE90_lastUpdate = performance.now();

function GE90_smoothFinalVolume() {
    const now = performance.now();
    const dt = (now - _GE90_lastUpdate) / 1000;
    _GE90_lastUpdate = now;

    const smoothing = 1 - Math.exp(-2.0 * dt);

    const base = window._GE90_customVolume;
    const atten = window._GE90_viewAttenuation || 1;
    const muted = window._GE90_muted ? 0 : 1;

    const target = base * atten * muted;

    if (window._GE90_finalMaster) {
        const g = window._GE90_finalMaster.gain;
        g.value += (target - g.value) * smoothing;
    }
}

        function toggleMute(){
          window._GE90_userMuted = !window._GE90_userMuted;
          window._GE90_persistentlyMuted = window._GE90_userMuted;
          applyEffectiveMute();
          showMuteIndicator();
          startContinuousMuteMonitoring();
          console.log("[GE90] enhanced user mute:", window._GE90_userMuted);
        }

        function togglePause(){
          window._GE90_paused = !window._GE90_paused;
          if (window._GE90_paused) {
            try { ctx.suspend(); } catch(e){}
          } else {
            try { ctx.resume(); } catch(e){}
          }
          applyEffectiveMute();
          console.log("[GE90] user pause:", window._GE90_paused);
        }

        window.addEventListener('keydown', e=>{
          try {
            if (e.key === 's' || e.key === 'S') {
              toggleMute();
            } else if (e.key === 'p' || e.key === 'P') {
              togglePause();
            }
          } catch(e){}
        });

        // -------------------------
        // Periodic view/mute updates + smooth volume sync
        // -------------------------
        setInterval(applyEffectiveMute, 250);
        // Note: GE90_smoothMasterVolume() is already running every 50ms from the custom slider section

        // Test slider identification (call manually to debug)
        window.debugGeoFSSliders = debugSliders;

        // Emergency functions
        window.emergencyMute = function() {
          console.log("[GE90] Emergency mute activated");
          window._GE90_persistentlyMuted = true;
          window._GE90_userMuted = true;
          forceMuteAllAudio();
          showMuteIndicator();
          startContinuousMuteMonitoring();
        };

        window.emergencyUnmute = function() {
          console.log("[GE90] Emergency unmute activated");
          window._GE90_persistentlyMuted = false;
          window._GE90_userMuted = false;
          showMuteIndicator();
          if (muteMonitoringInterval) {
            clearInterval(muteMonitoringInterval);
            muteMonitoringInterval = null;
          }
        };

        console.log("[GE90] page-context setup complete (v1.0) with persistent mute fix");
      } catch(e){
        console.warn("[GE90] setupPageContext failed", e);
      }
    });
  })();

})();
